class Property:

  def __init__(self):
    self.__name = ""

  @property
  def name(self):
    return self.__name
  @name.setter
  def name(self, value):
    if not isinstance(value, str):
      raise ValueError(
        "Name must be a string")
    if len(value) == 0:
      raise ValueError(
        "Name cannot be an empty string")
    self.__name = value
