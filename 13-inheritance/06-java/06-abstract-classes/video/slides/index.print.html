public class Vehicle{

  private String name;
  public double speed;

  public String getName(){ return this.name; }

  public Vehicle(String name){
    this.name = name;
    this.speed = 1.0;
  }

  public double move(double distance){
    System.out.println("Moving");
    return distance / this.speed;
  }

  public String describe(){
    return "";
  }

}
