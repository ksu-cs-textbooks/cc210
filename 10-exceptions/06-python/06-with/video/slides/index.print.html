import sys


try:
  if len(sys.argv) > 1:
    reader = open(sys.argv[1])
  else:
    reader = sys.stdin


  x = int(reader.readline())
  print(x)


except ValueError as e:
  print("ValueError: {}".format(e))
except FileNotFoundError as e:
  print("FileNotFoundError: {}".format(e))
finally:
  reader.close()
