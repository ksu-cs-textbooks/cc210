import java.util.Scanner;
import java.io.File;

public class Example{

  public static void main(String[] args){
    Scanner reader;

    if(args.length > 0){
      reader = new Scanner(new File(args[0]));
    }else{
      reader = new Scanner(System.in);
    }
    int x = reader.nextInt();
    System.out.println(x);

  }
}
